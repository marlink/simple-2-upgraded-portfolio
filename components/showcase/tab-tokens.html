/* Fluid Typography Example */
h1 { 
    font-size: clamp(2.5rem, 2rem + 2.5vw, 4rem);
    /* Scales from 40px to 64px */
}
.lead {
    font-size: clamp(1.125rem, 1.05rem + 0.375vw, 1.375rem);
    /* Scales from 18px to 22px */
}
